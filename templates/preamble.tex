% Theorem Environment Style

    \par\vspace{0.3cm}\noindent\textbf{#1}\par\vspace{0.1cm}
}{%
    \par\vspace{0.3cm}
}

% Definition Environment Style
\newenvironment{defbox}[1][]{%
    \par\vspace{0.35cm}\noindent\textbf{#1}\quad
}{%
    \par\vspace{0.25cm}
}

% Remark / Note / Example Environment Style

    \par\vspace{0.35cm}\noindent\textbf{#1}\par\vspace{0.15cm}
}{%

}
